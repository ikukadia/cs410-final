bool isHeightBalanced(TreeNode* root) {
  if (!root) {
    return true;
  }
  
  if (!root->left_  && root->right_) {
      return false;
    }
   else if (!root->right_ && root->left_) {
      return false;
    }

  if (heightHelper(root->left_) - heightHelper(root->right_) <= abs(1)) {
    return (isHeightBalanced(root->left_) == isHeightBalanced(root->right_));
  } else {
    return false;
  }
  
}

int heightHelper(TreeNode* root) {
  if (!root) {
    return -1;
  }

  int leftHeight = 1 + heightHelper(root->left_);
  int rightHeight = 1 + heightHelper(root->right_);

  return std::fmax(leftHeight, rightHeight);
}
